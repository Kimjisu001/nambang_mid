	user_pw	varchar2(25)		NOT NULL,
	user_name	varchar2(20)		NOT NULL,
	email	varchar2(50)		NOT NULL,
	phone	varchar2(15)		NOT NULL,
	gender	number		NULL,
	birth	date		NOT NULL,

);
